#Job search based on skill
* jobs|search for * jobs|show me * jobs|jobs related to *|jobs with * skill|jobs for *|* job|job for *
!example:Java jobs 
!expect:Java Developer, Location: Lehi, Utah, Apply Here
!console: 
{ 
"url":"https://jobs.github.com/positions.json?description=$1$",
"path":"$",
"actions":[{
    "type":"table",
    "columns":{"how_to_apply":"Apply","location":"Location","title":"Title"},
    "count":5
  }]
} 
eol
